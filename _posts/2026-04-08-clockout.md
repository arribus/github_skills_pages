### Calculate Your Shift
Enter your arrival time. This tool accounts for the **1h 15m mandatory break** (Total stay: 8h 15m).

<script>
    // --- LIVE CLOCK LOGIC ---
    function updateClock() {
        const now = new Date();
        const timeStr = now.toLocaleTimeString([], { hour12: false, hour: '2-digit', minute: '2-digit', second: '2-digit' });
        const dateStr = now.toLocaleDateString([], { weekday: 'long', year: 'numeric', month: 'long', day: 'numeric' });
        
        document.getElementById('live-clock').innerText = timeStr;
        document.getElementById('current-date').innerText = dateStr;
    }
    setInterval(updateClock, 1000);
    updateClock();

    // --- CALCULATION LOGIC ---
    const input = document.getElementById('arrival');
    const container = document.getElementById('result-container');

    input.addEventListener('input', () => {
        if (!input.value) {
            container.style.display = 'none';
            return;
        }
        
        const [hrs, mins] = input.value.split(':').map(Number);
        
        // Create Date objects for Arrival and Departure
        let arrivalDate = new Date();
        arrivalDate.setHours(hrs, mins, 0, 0);
        
        let departureDate = new Date(arrivalDate.getTime());
        // Add 8 hours and 15 minutes (495 minutes)
        departureDate.setMinutes(departureDate.getMinutes() + 495);

        // Format Displays
        const timeOptions = { hour: 'numeric', minute: '2-digit', hour12: true };
        const dateOptions = { weekday: 'short', month: 'short', day: 'numeric', year: 'numeric' };

        document.getElementById('arrival-date-display').innerText = 
            `${arrivalDate.toLocaleDateString([], dateOptions)} at ${arrivalDate.toLocaleTimeString([], timeOptions)}`;
        
        document.getElementById('departure-time').innerText = 
            departureDate.toLocaleTimeString([], timeOptions);
            
        document.getElementById('departure-date-display').innerText = 
            `Date: ${departureDate.toLocaleDateString([], dateOptions)}`;

        container.style.display = 'block';
    });
</script>
