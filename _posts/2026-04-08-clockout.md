<script>
    // Explicitly forcing 12-hour format across the board
    const fullTimeOptions = { hour: 'numeric', minute: '2-digit', second: '2-digit', hour12: true };
    const shortTimeOptions = { hour: 'numeric', minute: '2-digit', hour12: true };
    const dateDisplayOptions = { weekday: 'long', year: 'numeric', month: 'long', day: 'numeric' };

    // --- LIVE CLOCK ---
    function updateClock() {
        const now = new Date();
        // The 'en-US' locale forces AM/PM behavior even if the browser is in a 24h region (like UK/France)
        document.getElementById('live-clock').innerText = now.toLocaleTimeString('en-US', fullTimeOptions);
        document.getElementById('current-date').innerText = now.toLocaleDateString('en-US', dateDisplayOptions);
    }
    setInterval(updateClock, 1000);
    updateClock();

    // --- CALCULATION ---
    const input = document.getElementById('arrival');
    const container = document.getElementById('result-container');

    input.addEventListener('input', () => {
        if (!input.value) {
            container.style.display = 'none';
            return;
        }
        
        const [hrs24, mins] = input.value.split(':').map(Number);
        
        // Create an object for the arrival moment
        let arrivalDate = new Date();
        arrivalDate.setHours(hrs24, mins, 0, 0);
        
        // Create Departure Date (Arrival + 8h 15m)
        let departureDate = new Date(arrivalDate.getTime());
        departureDate.setMinutes(departureDate.getMinutes() + 495); // (8 * 60) + 15 = 495

        // Update UI with explicit 12h formatting
        document.getElementById('arrival-display').innerText = 
            `${arrivalDate.toLocaleDateString('en-US', {weekday: 'short', day: 'numeric'})} at ${arrivalDate.toLocaleTimeString('en-US', shortTimeOptions)}`;
        
        document.getElementById('departure-time').innerText = 
            departureDate.toLocaleTimeString('en-US', shortTimeOptions);
            
        document.getElementById('departure-date-display').innerText = 
            `Date: ${departureDate.toLocaleDateString('en-US', dateDisplayOptions)}`;

        container.style.display = 'block';
    });
</script>

---
**Why this works:**
1.  **Strict Locale:** By using `'en-US'` inside the `toLocaleTimeString` calls, we override the browser's regional settings. Even if you are in a country that uses 24h time, this dashboard will stay in **AM/PM**.
2.  **Live Precision:** The live clock updates every second, allowing you to count down to your exact badge-out moment.
3.  **Automatic Rollover:** If you badge in late at night, the "Departure Date" will correctly flip to the following day.
