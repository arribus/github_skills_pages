<script>
    // Configuration for 12-hour formatting
    const timeFormatOptions = { hour: 'numeric', minute: '2-digit', second: '2-digit', hour12: true };
    const shortTimeOptions = { hour: 'numeric', minute: '2-digit', hour12: true };
    const dateFormatOptions = { weekday: 'long', month: 'short', day: 'numeric', year: 'numeric' };

    // --- LIVE CLOCK ---
    function updateClock() {
        const now = new Date();
        document.getElementById('live-clock').innerText = now.toLocaleTimeString([], timeFormatOptions);
        document.getElementById('current-date').innerText = now.toLocaleDateString([], dateFormatOptions);
    }
    setInterval(updateClock, 1000);
    updateClock();

    // --- CALCULATION ---
    const input = document.getElementById('arrival');
    const container = document.getElementById('result-container');

    input.addEventListener('input', () => {
        if (!input.value) {
            container.style.display = 'none';
            return;
        }
        
        const [hrs24, mins] = input.value.split(':').map(Number);
        
        // Create Arrival Date
        let arrivalDate = new Date();
        arrivalDate.setHours(hrs24, mins, 0, 0);
        
        // Calculate Departure (Arrival + 8h 15m)
        let departureDate = new Date(arrivalDate.getTime());
        departureDate.setMinutes(departureDate.getMinutes() + 495); // 495 mins = 8h 15m

        // Display results in 12h format
        document.getElementById('arrival-display').innerText = 
            `${arrivalDate.toLocaleDateString([], {weekday: 'short', day: 'numeric'})} at ${arrivalDate.toLocaleTimeString([], shortTimeOptions)}`;
        
        document.getElementById('departure-time').innerText = 
            departureDate.toLocaleTimeString([], shortTimeOptions);
            
        document.getElementById('departure-date-display').innerText = 
            `Date: ${departureDate.toLocaleDateString([], dateFormatOptions)}`;

        container.style.display = 'block';
    });
</script>

### Why 12-hour format?
This dashboard is designed for quick readability during your internship. By using a consistent **AM/PM** format across the live clock and the tracker, you can immediately correlate your current status with your departure goal without performing 24-hour mental conversions.
